//https://leetcode.com/problems/lexicographically-smallest-palindrome/description/

class Solution {
public:
    string makeSmallestPalindrome(string s) {
        int i = 0, j = s.length()-1;
        while(i < j){
            if(s[i] != s[j]){
                char temp = min(int(s[i]),int(s[j]));
                s[i] = temp;
                s[j] = temp;
                
            }
            i++;
            j--;
        }
        return s;
    }
};
